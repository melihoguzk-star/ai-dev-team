SwiftUI — Renk Extension

ProteinOceanApp.swift 
extension Color {
    /// Hex string'den Color. "#2126AB" veya "2126AB" kabul eder.
    init(hex: String) {
        let hex = hex.trimmingCharacters(in: CharacterSet.alphanumerics.inverted)
        var int: UInt64 = 0
        Scanner(string: hex).scanHexInt64(&int)
        let r, g, b, a: UInt64
        switch hex.count {
        case 6:  (r, g, b, a) = (int >> 16, int >> 8 & 0xFF, int & 0xFF, 255)
        case 8:  (r, g, b, a) = (int >> 24, int >> 16 & 0xFF, int >> 8 & 0xFF, int & 0xFF)
        default: (r, g, b, a) = (0, 0, 0, 255)
        }
        self.init(.sRGB,
                  red:     Double(r) / 255,
                  green:   Double(g) / 255,
                  blue:    Double(b) / 255,
                  opacity: Double(a) / 255)
    }

    static let brandPrimary   = Color(hex: "#2126AB")  // Navigasyon, tab tint
    static let brandSecondary = Color(hex: "#232DA8")  // Fiyat rengi
    static let brandAccent    = Color(hex: "#1900FF")  // CTA border
    static let brandError     = Color(hex: "#DB2626")  // İndirim badge, hata
    static let brandWarning   = Color(hex: "#FF6600")  // Stok uyarısı, lansman
    static let brandBorder    = Color(hex: "#E5E5E5")  // Kart border, divider
}

enum Brand {
    enum Spacing {
        static let pagePadding: CGFloat = 20   // Figma: yatay padding
        static let gridGap: CGFloat     = 12   // Figma: ürün grid boşluğu
        static let sectionGap: CGFloat  = 16   // Bölüm arası boşluk
        static let touchMin: CGFloat    = 44   // iOS min touch target
    }

    enum Radius {
        static let page: CGFloat    = 20   // Figma: sayfa üst köşe
        static let card: CGFloat    = 12   // Ürün kartı, kategori kartı
        static let button: CGFloat  = 8    // Primary / secondary buton
        static let input: CGFloat   = 8    // Input alanı
        static let badge: CGFloat   = 6    // İndirim badge
    }

    enum Component {
        static let buttonHeight: CGFloat      = 56   // Figma
        static let inputHeight: CGFloat       = 50   // Figma
        static let productCardWidth: CGFloat  = 172  // Figma
        static let productCardHeight: CGFloat = 278  // Figma
        static let productImageHeight: CGFloat= 160  // Figma
    }
}
